8 {

13  GLuint texture;
14  glGenTextures(1, &texture);
15  glActiveTexture(GL_TEXTURE0 + unit);
16  glBindTexture(GL_TEXTURE_2D, texture);
17 
18  // choisit la representation des pixels de la texture en fonction de l'image
19  GLenum data_format= 0;
20  switch(im.channels)
21  {
22  case 1: data_format= GL_RED; break;
23  case 2: data_format= GL_RG; break;
24  case 3: data_format= GL_RGB; break;
25  case 4: data_format= GL_RGBA; break;
26  default: assert(0 && "invalid image format");
27  }
28 
29  // fixe les parametres de filtrage par defaut
30  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR);
31  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
32  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_BORDER);
33  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_BORDER);
34 
35  // transfere les donnees dans la texture
36  glTexImage2D(GL_TEXTURE_2D, 0,
37  GL_RGBA, im.width, im.height, 0,
38  data_format, GL_UNSIGNED_BYTE, &im.data.front());
