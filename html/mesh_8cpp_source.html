186 GLuint make_mesh_vertex_format( Mesh& m, const bool use_texcoord, const bool use_normal, const bool use_color )
187 {
188  if(m.positions.size() == 0)
189  return 0;
190 
191 #if 1
192  if(m.texcoords.size() > 0 && m.texcoords.size() < m.positions.size() && use_texcoord)
193  printf("[error] invalid texcoords array...\n");
194  if(m.normals.size() > 0 && m.normals.size() < m.positions.size() && use_normal)
195  printf("[error] invalid normals array...\n");
196  if(m.colors.size() > 0 && m.colors.size() < m.positions.size() && use_color)
197  printf("[error] invalid colors array...\n");
198 #endif
199 
200  // ne creer que les buffers necessaires
201  GLuint vao= create_vertex_format();
202  make_vertex_buffer(vao, 0, 3, GL_FLOAT, m.positions.size() * sizeof(vec3), &m.positions.front());
203  if(m.texcoords.size()== m.positions.size() && use_texcoord)
204  make_vertex_buffer(vao, 1, 2, GL_FLOAT, m.texcoords.size() * sizeof(vec2), &m.texcoords.front());
205  if(m.normals.size() == m.positions.size() && use_normal)
206  make_vertex_buffer(vao, 2, 3, GL_FLOAT, m.normals.size() * sizeof(vec3), &m.normals.front());
207  if(m.colors.size() == m.positions.size() && use_color)
208  make_vertex_buffer(vao, 3, 3, GL_FLOAT, m.colors.size() * sizeof(vec3), &m.colors.front());
209 
210  if(m.indices.size() > 0)
211  make_index_buffer(vao, m.indices.size() * sizeof(unsigned int), &m.indices.front());
212 
213  return vao;
214 }

264  program_use_texture(m.program, "diffuse_color", 0, texture);
